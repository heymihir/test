import pandas as pd

# Sample DataFrame (Replace with your actual DataFrame)
df = pd.DataFrame({
    'cust_id': [1, 2, 3],
    'RM': ['A', 'B', 'C'],
    'TL': ['X', 'Y', 'Z'],
    'region': ['North', 'South', 'East'],
    'city': ['NY', 'LA', 'CHI'],
    'EOP Revenue': [100, 200, 300],
    'MTD Sales': [400, 500, 600],
    'Avg Sales Mar': [700, 800, 900],
    'Avg Growth Feb': [1000, 1100, 1200],
    'Other Column': [1300, 1400, 1500]
})

# Conditions to filter columns
filter_conditions = [
    lambda col: 'EOP' in col or 'MTD' in col,  # Condition 1
    lambda col: ('Avg' in col and ('Mar' in col or 'Feb' in col)),  # Condition 2
    lambda col: col in ['cust_id', 'RM', 'TL', 'region', 'city']  # Condition 3
]

# Filter columns based on conditions
filtered_columns = [col for col in df.columns if any(cond(col) for cond in filter_conditions)]

# Create new DataFrame with filtered columns
filtered_df = df[filtered_columns]

# Display filtered DataFrame
import ace_tools as tools
tools.display_dataframe_to_user(name="Filtered DataFrame", dataframe=filtered_df)
