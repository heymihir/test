import pandas as pd

# Sample DataFrame with whitespace and mixed-case values
df = pd.DataFrame({
    ' col 1 ': [' A ', 'B ', ' C'],
    ' col 2 ': [' x ', ' Y', ' z '],
    'amount_os_lcy': [' 100 ', ' 200', '300 ']
})

# Remove whitespace from column names
df.columns = df.columns.str.strip().str.replace(' ', '_')

# Remove whitespace from all values and capitalize them
df = df.applymap(lambda x: str(x).strip().upper())

# Display the cleaned DataFrame
import ace_tools as tools
tools.display_dataframe_to_user(name="Cleaned DataFrame", dataframe=df)
