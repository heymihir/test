    # Food Processing & Consumer Food
    "food", "food ingredients", "flour", "sugar", "meal", "packaging", "package", "canned", "frozen",

# Example usage: check if a company name contains any PSL-related term
def is_psl_company(company_name):
    name = company_name.lower()
    return any(keyword in name for keyword in word_list)


patterns = [
    re.compile(r'\b' + re.escape(word.lower()) + r'(?:s|ing)?\b', re.IGNORECASE)
    for word in word_list
]

def sanitize(text):
    """Remove special characters except alphanumerics and whitespace."""
    return re.sub(r'[^a-zA-Z0-9\s]', '', text)

def is_psl(cust_name):
    """Check if the sanitized customer name contains any PSL keyword."""
    sanitized_name = sanitize(cust_name.lower())
    return any(pattern.search(sanitized_name) for pattern in patterns)

# Example usage with a DataFrame
df = pd.DataFrame({
    "cust_name": [
        "Jubliant Dairy",
        "Crazy Oil seeds",
        "Fast Package",
        "Delicious packaged food!",
        "Healthy Meals",
        "Healthy Mealing",
        "Crop Science Ltd. INC."
    ]
})

df["is_psl"] = df["cust_name"].apply(is_psl)
print(df)
