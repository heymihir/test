import pandas as pd

def update_base_data(base_data, new_data, base_key, new_key, columns_to_update, mapping):
    mask = base_data[base_key].isin(new_data[new_key])
    for col in columns_to_update:
        base_data.loc[mask, col] = base_data.loc[mask, base_key].map(
            new_data.set_index(new_key)[mapping[col]]
        )
    return base_data

df_updated = pd.DataFrame({
    'CUST_ID.1': [1, 2, 3, 4],
    'Name': ['Alice', 'Bob', 'Charlie', 'David'],
    'Amount': [100, 200, 300, 400],
    'Other': ['X', 'Y', 'Z', 'W']
})

df_new_customers = pd.DataFrame({
    'Cust Id': [2, 4],
    'New Name': ['Robert', 'Dave'],
    'New Amount': [250, 450],
    'Extra': ['A', 'B']
})

columns_to_update = ['Name', 'Amount']
mapping = {'Name': 'New Name', 'Amount': 'New Amount'}

df_updated = update_base_data(df_updated, df_new_customers, 'CUST_ID.1', 'Cust Id', columns_to_update, mapping)
print(df_updated)
