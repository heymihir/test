from pathlib import Path
from datetime import datetime, timedelta

def manage_folders(base_directory):
    base_dir = Path(base_directory)
    daily_dir = base_dir / "daily_balances_files"
    daily_dir.mkdir(parents=True, exist_ok=True)
    track_file = daily_dir / "last_run.txt"
    current_date = datetime.today().date()
    if track_file.exists():
        last_run_date = datetime.strptime(track_file.read_text().strip(), "%Y-%m-%d").date()
        if last_run_date < current_date:
            today_date = last_run_date
            track_file.write_text((current_date + timedelta(days=1)).strftime("%Y-%m-%d"))
        else:
            today_date = current_date
    else:
        today_date = current_date
        track_file.write_text((current_date + timedelta(days=1)).strftime("%Y-%m-%d"))
    tomorrow_date = current_date + timedelta(days=1)
    today_path = daily_dir / today_date.strftime("%Y-%m-%d")
    tomorrow_path = daily_dir / tomorrow_date.strftime("%Y-%m-%d")
    tomorrow_path.mkdir(parents=True, exist_ok=True)
    print(f'created folder: "{tomorrow_path}"')
    return today_path, tomorrow_path
